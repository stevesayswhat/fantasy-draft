<script>

const players = [

  {id:1, name:"Shohei Ohtani", team:"LAD", pos:"SP", pct:97.5, drafted:false},

  {id:2, name:"Juan Soto", team:"SD", pos:"OF", pct:89.1, drafted:false},

  {id:3, name:"Mookie Betts", team:"LAD", pos:"OF", pct:92.4, drafted:false},

  {id:4, name:"Ronald Acuña Jr.", team:"ATL", pos:"OF", pct:95.2, drafted:false}

];


let queue = [];

let pick = 1;

let round = 1;

let isMyTurn = true; // change to false to simulate waiting


function render() {

  document.getElementById("pick").innerText = pick;

  document.getElementById("round").innerText = round;


  const pool = document.getElementById("playerPool");

  pool.innerHTML = "";


  players.filter(p => !p.drafted).forEach(p => {

    const row = document.createElement("div");

    row.className = "row";


    row.innerHTML = `

      <span>${p.name}</span>

      <span>${p.team}</span>

      <span>${p.pos}</span>

      <span>${p.pct}%</span>

      <button class="queue-btn">Add to Queue</button>

      <button class="draft-btn ${isMyTurn ? "" : "hidden"}">DRAFT</button>

    `;


    row.querySelector(".queue-btn").onclick = () => {

      if (queue.length < 10 && !queue.includes(p.id)) {

        queue.push(p.id);

        render();

      }

    };


    row.querySelector(".draft-btn").onclick = () => {

      draftPlayer(p.id);

    };


    pool.appendChild(row);

  });


  const queueDiv = document.getElementById("queue");

  queueDiv.innerHTML = "";


  queue.forEach(id => {

    const p = players.find(pl => pl.id === id && !pl.drafted);

    if (!p) return;


    const row = document.createElement("div");

    row.className = "row";


    row.innerHTML = `

      <span>${p.name}</span>

      <button class="draft-btn ${isMyTurn ? "" : "hidden"}">DRAFT</button>

    `;


    row.querySelector(".draft-btn").onclick = () => {

      draftPlayer(p.id);

    };


    queueDiv.appendChild(row);

  });

}


function draftPlayer(id) {

  const player = players.find(p => p.id === id);

  if (!player || player.drafted) return;


  player.drafted = true;

  queue = queue.filter(q => q !== id);

  pick++;

  isMyTurn = false; // simulate turn ending

  render();

}


render();
